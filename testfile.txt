int find_min(int a[], int len){
    int min = a[0];
    int i = 0;
    while (i < len){
        if(a[i] < min){
            min = a[i];
        }
        i = i + 1;
    }
    return min;
}

int main(){
    int a[10];
    int i = 0;
    while (i < 10){
        a[i] = 10 - i;
        i = i + 1;
    }
    printf("%d\n",find_min(a, 10));
    return 0;
}
